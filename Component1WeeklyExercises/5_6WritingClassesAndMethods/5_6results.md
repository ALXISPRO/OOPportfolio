package week5_6;

public class Sphere {
	private double diameter;

	public Sphere(double diameter) {
        this.diameter = diameter;
    }
	
	public double getDiameter() {
        return diameter;
    }
	
	 
    public void setDiameter(double diameter) {
        this.diameter = diameter;
    }

    
    public double getVolume() {
        double radius = diameter / 2;
        return (4.0 / 3) * Math.PI * Math.pow(radius, 3);
    }
    
 
    public double getSurfaceArea() {
        double radius = diameter / 2;
        return 4 * Math.PI * Math.pow(radius, 2);
    }

    
    public String toString() {
        return "Diameter: " + diameter +
               ", Volume: " + getVolume() +
               ", Surface Area: " + getSurfaceArea();
    }
    
    

}

package week5_6;

public class Book {
	private String title, author, publisher;
    private int copyrightYear;

    public Book(String title, String author, String publisher, int year) {
        this.title = title;
        this.author = author;
        this.publisher = publisher;
        this.copyrightYear = year;
    }

    
    public String getTitle() { return title; }
    public void setTitle(String title) { this.title = title; }

    public String getAuthor() { return author; }
    public void setAuthor(String author) { this.author = author; }

    public String getPublisher() { return publisher; }
    public void setPublisher(String publisher) { this.publisher = publisher; }

    public int getYear() { return copyrightYear; }
    public void setYear(int year) { this.copyrightYear = year; }
    
    public String toString() {
        return "Title: " + title +
               "\nAuthor: " + author +
               "\nPublisher: " + publisher +
               "\nYear: " + copyrightYear + "\n";
    }
   

}

package week5_6;
import java.util.Random;

public class Cards {
	private String suit;
    private String face;

    public Cards() {
        String[] suits = {"Hearts", "Diamonds", "Clubs", "Spades"};
        String[] faces = {"Ace", "2", "3", "4", "5", "6", "7",
                          "8", "9", "10", "Jack", "Queen", "King"};
        Random rand = new Random();
        suit = suits[rand.nextInt(suits.length)];
        face = faces[rand.nextInt(faces.length)];
    }

    public String toString() {
        return face + " of " + suit;
    }

}


package week5_6;

public class DealCards {

	public static void main(String[] args) {
		for (int i = 1; i <= 5; i++) {
            Cards c = new Cards();
            System.out.println("Card " + i + ": " + c);
        }

	}

}
