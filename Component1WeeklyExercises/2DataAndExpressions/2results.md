import java.util.Scanner;

public class Average {

	public static void main(String[] args) {
		int val1 , val2 , val3;
		double average;
		Scanner scan = new Scanner (System.in);
		
		System.out.println("Please enter three integers and " + "I will compute their average");
		
		System.out.println("Enter the 1st value:");
		 val1 = scan.nextInt();
		System.out.println("Enter the 2nd value:");
		 val2 = scan.nextInt();
		System.out.println("Enter the 3rd value:");
		 val3 = scan.nextInt();

		System.out.println("The average is:" + average);
		
		scan.close();
		
		

	}

}

package weektwo;

import java.util.Scanner;

public class fractions {

	public static void main(String[] args) {
			int numerator;
			int denominator;
			double decimalvalue;
			
			Scanner scan = new Scanner(System.in);
			
			System.out.println("Enter the numerator: ");
			numerator = scan.nextInt();
			
			System.out.println("Enter the denominator:");
			denominator = scan.nextInt();
			
			if (denominator==0)
			{
				System.out.println("Error:Denominator cannot be zero");
				
			}
			else
			{
				decimalvalue=(double)numerator/denominator;
				System.out.println("The decimal value is :" + decimalvalue);
				
			}
			scan.close();
	}

}
